async function uploadModels(models) {
    const token = "YOUR_GITHUB_TOKEN";
    const repo = "swapnillad8/Product-Sourcing-Portal";

    const content = btoa(models.join("\n"));

    await fetch(`https://api.github.com/repos/${repo}/contents/models.txt`, {
        method: "PUT",
        headers: {
            "Authorization": `Bearer ${token}`,
            "Accept": "application/vnd.github+json"
        },
        body: JSON.stringify({
            message: "Update models",
            content: content
        })
    });
}
